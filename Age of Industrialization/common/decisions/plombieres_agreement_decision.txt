plombieres_agreement_decision = {
	is_shown = {	
		NOT = { 
			has_variable = plombieres_agreement_var 
		}
        exists = c:SAR
		c:SAR = ROOT
	}

	possible = {
		is_at_war = no
		country_rank >= rank_value:minor_power
		is_subject = no
		is_at_war = no
		has_technology_researched = nationalism
		owns_entire_state_region = STATE_PIEDMONT
		owns_entire_state_region = STATE_SAVOY
		has_state_in_state_region = STATE_PROVENCE
		OR = {
			c:AUS = {
				owns_entire_state_region = STATE_LOMBARDY
				is_subject = no
			}
			any_country = {
				is_subject_of = c:AUS
				owns_entire_state_region = STATE_LOMBARDY
			}
		}
		year >= 1850
		NOT = {
			year >= 1870
		}
		c:FRA = {
			relations:root >= relations_threshold:amicable 
			is_at_war = no
		}
		c:SAR = {
			is_at_war = no
		}
	}

	immediate = {
		c:FRA = {
			save_scope_as = fra_scope
		}
	}

	when_taken = {
		set_variable = {
			name = plombieres_agreement_var
			value = yes
		}
		trigger_event = {
			id = plombieres_agreement.0
			days = 0
			popup = yes
		}
	}

	ai_chance = {
		base = 1
		modifier = {
			trigger = { 
				weekly_net_fixed_income > money_amount_multiplier_medium
				gold_reserves > 0
			}
			add = 1
		}
		modifier = {
			trigger = {
				gold_reserves < -350000 
			}
			add = -1000
			}	
		modifier = {
			trigger = {
				has_strategy = ai_strategy_armed_isolationism
			}
			add = -1000
			}		
		modifier = {
			OR = {
				is_diplomatic_play_committed_participant = yes
				is_at_war = yes
			}
			add = -1000
		}			
	}

	weight = 500
}
