je_lca_gold_reserves_limit = {
	icon = "gfx/interface/icons/event_icons/event_trade.dds"

	is_shown_when_inactive = {
		gdp > 1000000000
	}

	possible = {
		gdp > 2000000000
	}

	immediate = {
		if  = {
			limit = {
				NOT = { has_modifier = lca_gold_reserves_limit_modifier }
			}
			add_modifier = {
				name = lca_gold_reserves_limit_modifier
			}
		}
		if = {
			limit = {
				has_modifier = ig_national_dividends
			}
			if = {
				limit = { NOT = { has_modifier = lca_gold_reserves_limit_modifier_2 } }
				add_modifier = {
					name = lca_gold_reserves_limit_modifier_2
				}
			}
		}
		else = {
			if = {
				limit = { has_modifier = lca_gold_reserves_limit_modifier_2 }
				remove_modifier = lca_gold_reserves_limit_modifier_2
			}
		}
	}

	on_weekly_pulse = {
		effect = {
			if  = {
				limit = {
					NOT = { has_modifier = lca_gold_reserves_limit_modifier }
				}
				add_modifier = {
					name = lca_gold_reserves_limit_modifier
				}
			}
			if = {
				limit = {
					has_modifier = ig_national_dividends
				}
				if = {
					limit = { NOT = { has_modifier = lca_gold_reserves_limit_modifier_2 } }
					add_modifier = {
						name = lca_gold_reserves_limit_modifier_2
					}
				}
			}
			else = {
				if = {
					limit = { has_modifier = lca_gold_reserves_limit_modifier_2 }
					remove_modifier = lca_gold_reserves_limit_modifier_2
				}
			}
		}
	}

	complete = {
		always = no
	}

	on_complete = {
		if = {
			limit = { has_modifier = lca_gold_reserves_limit_modifier }
			remove_modifier = lca_gold_reserves_limit_modifier
		}
		if = {
			limit = { has_modifier = lca_gold_reserves_limit_modifier_2 }
			remove_modifier = lca_gold_reserves_limit_modifier_2
		}
	}
}
