je_acw_countdown = {
	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	immediate = {
		set_variable = {
			name = acw_progress_var
			value = 0
		}
	}

	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
		has_law = law_type:law_slavery_banned
	}

	on_complete = {
		remove_modifier = missouri_compromise
	}

	fail = {
		OR = {
			# CSA via outlawing slavery
			custom_tooltip = {
			text = acw_outlawing_slavery
				c:USA = {
					is_enacting_law = law_type:law_slavery_banned
				}
			}
			# CSA via abolitionist president
			custom_tooltip = {
			text = acw_abolitionist_president
				c:USA = {
					ruler = {
						has_ideology = ideology_abolitionist
					}
				}
				year > 1845
			}
			ig:ig_landowners = {
				is_marginal = no
				ig_approval < -19
			}
		}
	}

	on_fail = {
		custom_tooltip = {
			text = acw_will_happen_tt
			if = {
				limit = {
					OR = {
						c:USA = {
							ruler = {
								has_ideology = ideology_abolitionist
							}
						}
						c:USA = {
							is_enacting_law = law_type:law_slavery_banned
						}
						c:USA = {
							ig:ig_landowners = {
								is_marginal = no
								ig_approval < -19
							}
						}
					}
				}
				set_variable = csa_will_spawn_var
			}
			trigger_event = { id = acw_je_events.1 }
		}
	}

	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					has_law = law_type:law_slavery_banned
				}
				change_variable = {
					name = acw_progress_var
					add = 1
				}
			}
		}
		random_events = {
			# American Civil War buildup
			100 = 0
			20 = acw_events.1 # The Abolitionist's Martyrdom
			20 = acw_events.3 # The Wilmot Proviso
			100 = acw_sectionalism_startup.1 #Growing Sectionalism
		}
	}
	on_yearly_pulse = {
		random_events = {
			40 = 0
			20 = acw_events.2 # The Fugitive Slave
			20 = acw_events.5 # Bleeding Kansas
			20 = acw_events.6 # The Caning of Charles Sumner
			20 = acw_sectionalism_startup.2 #Dredd Scott Case
		}
	}

	current_value = {
		value = root.var:acw_progress_var
	}

	goal_add_value = {
		value = 120
	}

	progressbar = yes

	weight = 100

	should_be_pinned_by_default = yes
}

	complete = {
		custom_tooltip = {
			text = acw_war_tt_complete
			c:CSA = {
			is_subject_of = c:USA
			}
		}
	}

	on_complete = {
		if = {
			limit = {
				has_variable = csa_will_spawn_var
			}

		}
		else = {
		trigger_event = { id = acw_peace_treaty.3 }
		}
	}
	
	fail = {

	weight = 10000

	# acw_events.7 fires for the FSA or CSA, but requires the je_acw_war journal to be active

je_acw_reconstruction = {
	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
	}

	on_complete = {
		trigger_event = { id = acw_reconstruction.2 }
		hidden_effect = { # sanity check just in case something somehow falls through the cracks
			every_scope_state = {
				limit = { has_modifier = unreconstructed_state }
				remove_modifier = unreconstructed_state
			}
		}
	}

	current_value = {
		value = var:reconstruction_var
	}

	goal_add_value = {
		add = 3
	}

	timeout = 4380

	should_be_pinned_by_default = yes
	progressbar = yes

	on_timeout = {
		trigger_event = { id = acw_reconstruction.2 }
	}
}

je_acw_reincorporate = {
	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	complete = {
		NOT = {
			any_scope_state = {
				has_modifier = unreconstructed_state
				is_incorporated = no
			}
		}
		every_scope_state = {
			limit = {
				is_homeland = cu:dixie
				has_modifier = unreconstructed_state
			}
				loyalist_fraction = {
					value > 0.10
					culture = cu:dixie
				}
				devastation < 0.05
		}
	}

	on_complete = {
		custom_tooltip = {
			text = dont_lose_dixie_culture
			set_variable = acw_keep_dixie
		}
		custom_tooltip = {
			text = reconstruction_progress
			change_variable = {
				name = reconstruction_var
				add = 1
			}
		}
	}

	on_monthly_pulse = {
		random_events = {
			50 = 0
			20 = acw_followup_events.6
			20 = acw_followup_events.7
			20 = acw_followup_events.8
		}
	}

	invalid = {
		NOT = {
			any_scope_state = {
				state_region = {
					is_homeland = cu:dixie
				}
			}
		}
	}

	timeout = 4380
	should_be_pinned_by_default = yes
}
