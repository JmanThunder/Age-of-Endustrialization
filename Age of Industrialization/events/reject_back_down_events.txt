namespace = reject_back_down_event

# setup target if target exists and isn't now a subject of root
reject_back_down_event.1 = {
	type = diplo_play_event
	placement = root
	hidden = yes

	immediate = {
		target = {
			root = {
				set_variable = {
					name = temp
					value = prev
				}
			}
		}
		if = {
			limit = {
				initiator = {
					has_variable = target
				}
			}
			initiator = {
				remove_variable = target
			}
		}
		if = {
			limit = {
				initiator = {
					is_player = yes
				}
				exists = var:temp
				var:temp = {
					any_scope_state = {
						is_capital = yes
					}
				}
			}
			initiator = {
				set_variable = {
					name = target
					value = root.var:temp
				}
			}
		}
	}
}

	}

	option = {

			}
		}
		remove_variable = target
	}
}
