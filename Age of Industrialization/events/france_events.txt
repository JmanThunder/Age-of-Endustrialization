napoleonic_flavor.1 = {
	type = country_event
	placement = ROOT

	event_image = {
		video = "gfx/event_pictures/europenorthamerica_springtime_of_nations.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/europenorthamerica/springtime_of_nation"

	icon = "gfx/interface/icons/event_icons/event_protest.dds"
	
	title = napoleonic_flavor.1.t
	desc = napoleonic_flavor.1.d
	flavor = napoleonic_flavor.1.f

	duration = 3

	cooldown = { months = normal_modifier_time }
	
	trigger = {
        exists = c:FRA
        c:FRA = ROOT
		has_law = law_type:law_monarchy
		NOT = { has_variable = french_empire_created }
		OR = {
			is_enacting_law = law_type:law_presidential_republic
			is_enacting_law = law_type:law_parliamentary_republic
		}
	}
	
	immediate = {
	}
	
	option = {
		name = napoleonic_flavor.1.a
		if = { 
			limit = {
				has_law = law_type:law_presidential_republic
			}
			custom_tooltip = napoleonic_flavor.1.a.tt
		}
		hidden_effect = {
			if = { 
				limit = {
					has_law = law_type:law_parliamentary_republic
				}
				set_variable = louis_napoleon_created
				create_character = {
					first_name = Louis-Napoleon
					last_name = Bonaparte
					ig_leader = yes
					interest_group = ig_armed_forces
					female = no
					birth_date = 1808.4.20
					dna = dna_emperor_napoleon_03
					ideology = ideology_moderate
					traits = {
						ambitious
						demagogue
					}
					on_created = {
						set_variable = is_louis_napoleon
					}
				}
			}
			if = { 
				limit = {
					has_law = law_type:law_presidential_republic
				}
				trigger_event = { id = napoleonic_flavor.6 days = 50 popup = yes }
			}
		}
	}
}

royalist_restauration.1 = {
	type = country_event
	placement = ROOT

	event_image = {
		video = "gfx/event_pictures/europenorthamerica_london_center.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/europenorthamerica/london_center"

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"
	
	title = {
		first_valid = {
			triggered_desc = {
				trigger = {
					NOT = { has_variable = henri_restauration }
				}
				desc = royalist_restauration.1.t1
			}
			triggered_desc = {
				trigger = {
					always = yes
				}
				desc = royalist_restauration.1.t2
			}
		}
	}
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = {
					NOT = { has_variable = henri_restauration }
				}
				desc = royalist_restauration.1.d1
			}
			triggered_desc = {
				trigger = {
					always = yes
				}
				desc = royalist_restauration.1.d2
			}
		}
	}
	flavor = {
		first_valid = {
			triggered_desc = {
				trigger = {
					NOT = { has_variable = henri_restauration }
				}
				desc = royalist_restauration.1.f1
			}
			triggered_desc = {
				trigger = {
					always = yes
				}
				desc = royalist_restauration.1.f2
			}
		}
	}

	duration = 3

	cooldown = { months = normal_modifier_time }
	
	trigger = {
        exists = c:FRA
        c:FRA = ROOT
		is_enacting_law = law_type:law_monarchy
		ruler = {
			NOT = { has_variable = is_louis_napoleon }
			has_ideology = ideology_legitimist
		}
	}
	
	immediate = {
	}
	
	option = {
		name = royalist_restauration.1.a
		hidden_effect = {
			if = { 
				limit = {
					NOT = { has_variable = henri_restauration }
				}
				create_character = {
					first_name = "Henri"
					last_name = "de_BorbO_n"
					ruler = yes
					birth_date = 1820.9.29
					dna = dna_king_henry_05
					interest_group = ig_landowners
					ideology = ideology_legitimist
					traits = {
						reserved
					}
					on_created = {
						set_variable = is_married
						set_variable = is_legitimist
					}
				}
				set_variable = henri_restauration
			}
			else = {
				create_random_legitimist_ruler = yes
			}
		}
	}
}
