napoleonic_flavor.1 = {
	type = country_event
	placement = ROOT

	event_image = {
		video = "gfx/event_pictures/europenorthamerica_springtime_of_nations.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/europenorthamerica/springtime_of_nation"

	icon = "gfx/interface/icons/event_icons/event_protest.dds"
	
	title = napoleonic_flavor.1.t
	desc = napoleonic_flavor.1.d
	flavor = napoleonic_flavor.1.f

	duration = 3

	cooldown = { months = normal_modifier_time }
	
	trigger = {
        exists = c:FRA
        c:FRA = ROOT
		has_law = law_type:law_monarchy
		NOT = { has_variable = french_empire_created }
		OR = {
			is_enacting_law = law_type:law_presidential_republic
			is_enacting_law = law_type:law_parliamentary_republic
		}
	}
	
	immediate = {
	}
	
	option = {
		name = napoleonic_flavor.1.a
		hidden_effect = {
			create_character = {
				first_name = Louis-Napoleon
				last_name = Bonaparte
				ig_leader = yes
				interest_group = ig_armed_forces
				female = no
				birth_date = 1808.4.20
				dna = dna_emperor_napoleon_03
				ideology = ideology_moderate
				traits = {
					ambitious
					demagogue
				}
				on_created = {
					set_variable = is_louis_napoleon
				}
			}
		}
	}
}
